  <div class="player" role="region" aria-label="Минималистичный аудио плеер">

    <button class="play" id="playBtn" aria-pressed="false" title="Play / Stop">

      <svg id="icon" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.6" stroke-linecap="round" stroke-linejoin="round">

        <path d="M5 3v18l15-9L5 3z"></path>

      </svg>

    <input id="vol" class="vol" type="range" min="0" max="1" step="0.01" value="1" aria-label="Громкость" />

  </div>


  <audio id="audio" crossorigin="anonymous" src="untitled_.mp3"></audio>


  <script>

    const audio = document.getElementById('audio');

    const playBtn = document.getElementById('playBtn');

    const icon = document.getElementById('icon');

    const vol = document.getElementById('vol');


    playBtn.addEventListener('click', async () => {

      try {

        if(audio.paused){

          await audio.play();

          playBtn.setAttribute('aria-pressed','true');

          icon.innerHTML = '<path d="M6 19h4V5H6v14zm8-14v14h4V5h-4z"></path>';

        } else {

          audio.pause();

          playBtn.setAttribute('aria-pressed','false');

          icon.innerHTML = '<path d="M5 3v18l15-9L5 3z"></path>';

        }

      } catch(err){

        console.warn('Playback error', err);

      }

    });


    vol.addEventListener('input', ()=>{

      audio.volume = parseFloat(vol.value);

    });


    audio.addEventListener('play', ()=>{

      playBtn.setAttribute('aria-pressed','true');

      icon.innerHTML = '<path d="M6 19h4V5H6v14zm8-14v14h4V5h-4z"></path>';

    });

    audio.addEventListener('pause', ()=>{

      playBtn.setAttribute('aria-pressed','false');

      icon.innerHTML = '<path d="M5 3v18l15-9L5 3z"></path>';

    });

  </script>
